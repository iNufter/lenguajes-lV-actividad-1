    private void jbtncalcularActionPerformed(java.awt.event.ActionEvent evt) {                                             
        // TODO add your handling code here:
        String peso = jtfpeso.getText();
        String estatura = jtfestatura.getText();
        
        float p = Float.parseFloat(peso);
        float e = Float.parseFloat(estatura);
        
        persona miPersona = new persona ();
        miPersona.setPeso(p);
        miPersona.setEstatura(e);
        miPersona.setImc();
        
        if(miPersona.getImc() < 18.5){
            JOptionPane.showMessageDialog(null, "BAJO PESO");
        } else if(miPersona.getImc() >= 18.5 && miPersona.getImc() <= 24.9){
            JOptionPane.showMessageDialog(null, "NORMAL");
        }else if(miPersona.getImc() >= 25 && miPersona.getImc() <= 29.9){
            JOptionPane.showMessageDialog(null, "SOBREPESO");
        }else if(miPersona.getImc() >= 30 && miPersona.getImc() <= 34.9){
            JOptionPane.showMessageDialog(null, "OBESIDAD I");
        }else if(miPersona.getImc() >= 35 && miPersona.getImc() <= 39.9){
            JOptionPane.showMessageDialog(null, "OBESIDAD II");
        } else {
            JOptionPane.showMessageDialog(null, "OBESIDAD III");
        }
    
       jtfpeso.setText("");
       jtfestatura.setText("");
        
        
    }
